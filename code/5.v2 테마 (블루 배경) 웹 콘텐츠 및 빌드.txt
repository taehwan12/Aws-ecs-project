# 작업 디렉토리 생성
mkdir -y ~/nginx-v2 && cd ~/nginx-v2

# index.html 파일 생성 (블루 테마)
cat << 'EOF' > index.html
<!DOCTYPE html>
<html>
<head>
    <title>AWS ECS Project - v2</title>
    <style>
        body { background-color: #004488; color: white; font-family: sans-serif; text-align: center; padding-top: 100px; }
    </style>
</head>
<body>
    <h1>🚀 AWS ECS 가동 인프라 - Version 2 (Blue Theme)</h1>
    <p>Managed by 4RSENAL Infatructure Team</p>
</body>
</html>
EOF

# Dockerfile 생성 (동일)
cat << 'EOF' > Dockerfile
FROM nginx:latest
COPY index.html /usr/share/nginx/html/index.html
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
EOF

# v2 이미지 로컬 빌드
docker build -t ecs-nginx:v2 .
